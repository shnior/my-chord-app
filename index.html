let audioContext; let source; let analyzer; const startBtn = document.getElementById('start-btn'); const audioFile = document.getElementById('audio-file'); const audioPlayer = document.getElementById('audio-player'); const chordResult = document.getElementById('chord-result'); // פונקציה לעדכון מצב הכפתור function updateButtonState() { if (audioFile.files.length > 0) { startBtn.disabled = false; chordResult.innerText = "מוכן!"; } } audioFile.onchange = (e) => { if (e.target.files[0]) { audioPlayer.src = URL.createObjectURL(e.target.files[0]); audioPlayer.style.display = 'block'; updateButtonState(); } }; // בדיקה ראשונית אם כבר נבחר קובץ (למקרה של ריענון) updateButtonState(); startBtn.onclick = async () => { if (typeof Meyda === 'undefined') { alert("הספרייה Meyda לא נטענה. אנא המתן או רענן."); return; } try { if (!audioContext) { audioContext = new (window.AudioContext || window.webkitAudioContext)(); } await audioContext.resume(); if (!source) { source = audioContext.createMediaElementSource(audioPlayer); source.connect(audioContext.destination); } if (analyzer) analyzer.stop(); analyzer = Meyda.createMeydaAnalyzer({ audioContext: audioContext, source: source, bufferSize: 1024, featureExtractors: ['chroma'], callback: features => { features.chroma.forEach((val, i) => { const bar = document.getElementById(`bar-${i}`); if(bar) bar.style.height = (val * 100) + '%'; }); let bestChord = "---"; let maxSim = 0.5; const CHORD_TEMPLATES = { 'C Major': [0, 4, 7], 'C# Major': [1, 5, 8], 'D Major': [2, 6, 9], 'Eb Major': [3, 7, 10], 'E Major': [4, 8, 11], 'F Major': [5, 9, 0], 'F# Major': [6, 10, 1], 'G Major': [7, 11, 2], 'Ab Major': [8, 0, 3], 'A Major': [9, 1, 4], 'Bb Major': [10, 2, 5], 'B Major': [11, 3, 6], 'C Minor': [0, 3, 7], 'D Minor': [2, 5, 9], 'E Minor': [4, 7, 11], 'A Minor': [9, 0, 4], 'G Minor': [7, 10, 2] }; for (const [name, notesIdx] of Object.entries(CHORD_TEMPLATES)) { let sim = notesIdx.reduce((sum, idx) => sum + features.chroma[idx], 0); if (sim > maxSim) { maxSim = sim; bestChord = name; } } chordResult.innerText = bestChord; } }); analyzer.start(); audioPlayer.play(); startBtn.innerText = "מנתח..."; startBtn.disabled = true; } catch (err) { alert("שגיאה: " + err.message); } };
